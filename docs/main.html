    Make a call to the github API. The path can be either a relative path such as users/STRd6 or an absolute path like https://api.github.com/users/octocat or user.url.

    We attach our accessToken if present.

    api returns a promise for easy chaining.

      api = (path, options={}) ->
    if path.match /^http/
      url = path
    else
      url = "https://api.github.com/#{path}"

    options.headers ||= {}

    if token
      options.headers["Authorization"] = "token #{token}"

    options = extend
      url: url
      type: "GET"
      dataType: 'json'
      contentType: "application/json; charset=utf-8"
    , options
